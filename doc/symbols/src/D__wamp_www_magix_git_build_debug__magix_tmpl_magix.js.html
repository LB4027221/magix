  2 /**
  3  * 检测某个对象是否拥有某个属性
  4  * @param  {Object}  owner 检测对象
  5  * @param  {String}  prop  属性
  6  * @return {Boolean} 是否拥有prop属性 
  7  */
  8 var has=function(owner,prop){
  9 	if(!owner)return false;//false 0 null '' undefined
 10 	return Object.prototype.hasOwnProperty.call(owner,prop);
 11 };

 13 /**
 14  * 混合对象的属性
 15  * @param  {Object} aim    要mix的目标对象
 16  * @param  {Object} src    mix的来源对象
 17  * @param  {Object} ignore 在复制时，忽略的值
 18  * @return {Object}
 19  */
 20 var mix=function(aim,src,ignore){
 21 	for(var p in src){
 22 		if(has(src,p)&&(!ignore||!has(ignore,p))){
 23 			aim[p]=src[p];
 24 		}
 25 	}
 26 	return aim;
 27 };

 29 /**
 30  * 以try cache方式执行方法，忽略掉任何异常
 31  * @param  {Array} fns     函数数组
 32  * @param  {Array} args    参数数组
 33  * @param  {Object} context 在待执行的方法内部，this的指向
 34  * @return {Object} 返回执行的最后一个方法的返回值
 35  */
 36 var safeExec=function(fns,args,context,i,r,e){
 37 	if(!Magix.isArray(fns)){
 38 		fns=[fns];
 39 	}
 40 	if(!args||(!Magix.isArray(args)&&!args.callee)){
 41 		args=[args];
 42 	}
 43 	for(i=0;i<fns.length;i++){
 44 		try{
 45 			e=fns[i];
 46 			r=Magix.isFunction(e)&&e.apply(context,args);
 47 		}catch(x){
 48 			S.log(x,e);
 49 		}
 50 	}
 51 	return r;
 52 };
